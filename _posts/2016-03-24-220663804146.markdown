varchar() = 대소문자 구분 안함

varbinary() = 대소문자 구분

mysql> SELECT * FROM TOKEN_DIC WHERE token_name  = 'icon';

+------------+

| token_name |

+------------+

| icon       |

| ICON       |

+------------+

2 rows in set (0.51 sec)

mysql> SELECT * FROM TOKEN_DIC WHERE token_name  = binary( 'icon' );

+------------+

| token_name |

+------------+

| icon       |

+------------+

1 row in set (0.52 sec)
