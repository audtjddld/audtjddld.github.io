        function converted(str){
            //alert(str);
            var value=str.split(':');
            var name=trim(value[0]);
            var talk=trim(value[1]);
            cnt++;

            if(name!="" && talk !=""){
                    return "<tr><td bgcolor='#f5f5f5'>"+value[0]+"</td> <td>"+value[1]+"</td></tr>";
            }
        }





$(document).ready(function() {
            $.ajax({
                    type :"post",
                    url : "/????/xml/data_01_01.xml", // xml 경로
                    dataType: "xml", // xml 타입
                    success : function(xml) {
                       /**/
                        var idx =$(xml).find("start").children().length+1; //elements 갯수
                        var engIdx=0; // 영어 문장 갯수
                        var korIdx=0; // 국어 문장 갯수
                        var lineIdx=0; // 가로줄 갯수

                        var htmlSrc = new Array();

                        for(var i =1; i<idx;i++){
                            engIdx=$(xml).find("e"+i).children().length+1;
                                for(var j=1; j<engIdx; j++ ){
                                    htmlSrc.push(converted($(xml).find("e"+i).find("talk"+j).text()));
                                }
                            korIdx=$(xml).find("k"+i).children().length+1;
                                for(var h=1; h<korIdx; h++){
                                    htmlSrc.push(converted($(xml).find("k"+i).find("talk"+h).text()));
                                }
                            if(lineIdx==0){ htmlSrc.push("<tr><td colspan='2'><hr></td></tr>"); lineIdx++;}
                            $("#xmlGrid tbody").html(htmlSrc);
                        }
                        /**/
                    }
            });
        });
