    function randomPassword()

    {

    var  chars = "abcdefghijklmnopqrstuvwxyz1234567890";

          pass = "";

      for(var x=0;x<10;x++)

      {

        i = Math.floor(Math.random() * 36);

        pass += chars.charAt(i);

      }

      return pass;

    }
