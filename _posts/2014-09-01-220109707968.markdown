// JavaScript Code



$(document).ready(function(){

    var map = new Map();



    map.put("zzz","123");



    alert(map.get("zzz"));

});



function Map(){

    this.map = new Object();

}



Map.prototype = {

    put : function(key, value) {

        this.map[key] = value;

    },

    get : function(key) {

        return this.map[key];

    }

}
