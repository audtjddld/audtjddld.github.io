SELECT LTRIM(RTRIM(OWNER)) || '.' || OBJECT_NAME OBJ_NAME, OBJECT_TYPE,

DECODE(OBJECT_TYPE, 'TABLE', 'DROP TABLE ' ||

OBJECT_NAME || ';', '') DROP_TABLE

FROM ALL_OBJECTS

WHERE OWNER LIKE UPPER('%icpa%')

AND OBJECT_TYPE = 'TABLE'

ORDER BY OBJECT_NAME;
