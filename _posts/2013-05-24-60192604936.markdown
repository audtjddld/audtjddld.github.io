<script type="text/javascript">

 function Up(obj){
  var list = document.getElementById(obj);
  var index = list.selectedIndex;
  var saveText, saveValue;

  if(index<0){
   alert("메뉴를 선택하세요");
  }else if(index==0){
   alert("더이상 위로 이동이 불가합니다");
  }else{
   saveText  = list[index].text;
   saveValue = list[index].value;

   list[index]=null;

   var item = new Option(saveText, saveValue);

   list.options.add(item,index - 1);
   list.selectedIndex=index - 1;
  }
 }

 function Down(obj){
  var list = document.getElementById(obj);
  var index = list.selectedIndex;
  var maxLength = list.length;
  var saveText, saveValue;

  if(index<0){
   alert("메뉴를 선택하세요");
  }else if(index==maxLength-1){
   alert("더이상 내릴 수 없습니다.");
  }else{
   saveText  = list[index].text;
   saveValue = list[index].value;

   list[index]=null;

   var item = new Option(saveText, saveValue);

   list.options.add(item,index + 1);
   list.selectedIndex=index + 1;
  }
 }
</script>
