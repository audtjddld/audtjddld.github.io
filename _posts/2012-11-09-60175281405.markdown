<c:forEach items="${fn:split(value,',')}" var="icon">
         <c:if test="${icon=='1'}">
             <img src="이미지" width="28" height="11" alt="new" />
         </c:if>
          <c:if test="${icon=='2'}">
              <img src="이미지" width="48" height="11" alt="upgrade" />
          </c:if>
         <c:if test="${icon=='4'}">
          </c:if>
         <c:if test="${icon=='8'}">
           </c:if>
         <c:if test="${icon=='16'}">
              <img src="이미지" width="31" height="11" alt="best" />
          </c:if>
</c:forEach>
