SELECT SUM(DECODE(YO,1,0,7,0,1)) RESULT FROM
(
 SELECT TO_CHAR(SYSDATE+(LEVEL-365*5),'YYYYMMDD') DATES,TO_CHAR(SYSDATE+(LEVEL-365*5),'D') YO FROM DUAL
 CONNECT BY LEVEL-365*5< 365 * 5 + 1
)
WHERE DATES BETWEEN '20070715' AND '20070731'
